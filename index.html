import os
import json
import time
import requests
import folium
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter
from dotenv import load_dotenv
from openai import OpenAI

# Get the OpenAI API key from the .env file
load_dotenv('.env', override=True)
openai_api_key = os.getenv('OPENAI_API_KEY')

# Set up the OpenAI client
client = OpenAI(api_key=openai_api_key)


def execute_llm_call(prompt, model="gpt-5.5"):
    """
    Internal execution handler that dynamically detects the correct OpenAI endpoint architecture,
    binds appropriate token limits, configures tool payloads, and cleanly extracts output string data.
    Includes a buffer-flushed real-time polling loop for long-running deep-research agent requests.
    """
    if not isinstance(prompt, str):
        raise ValueError("Input must be a string enclosed in quotes.")

    messages_context = [
        {
            "role": "system",
            "content": "You are a helpful but terse AI assistant who gets straight to the point.",
        },
        {"role": "user", "content": prompt},
    ]

    # Explicit routing logic for primary API selection
    prefer_responses_api = (model.startswith("gpt-5") or "deep-research" in model or model == "o3-pro") and not (model == "gpt-5-search-api")

    if prefer_responses_api:
        try:
            kwargs = {
                "model": model,
                "input": messages_context,
                "max_output_tokens": 4096
            }
            
            # Special logic for agentic deep-research execution
            if "deep-research" in model:
                kwargs["tools"] = [{"type": "web_search_preview"}]
                kwargs["background"] = True  
                
                initial_job = client.responses.create(**kwargs)
                job_id = initial_job.id
                
                print(f"⚡ [Deep Research Engine] Initializing multi-step agent workflow for {model}...", flush=True)
                start_time = time.time()
                
                while True:
                    current_run = client.responses.retrieve(job_id)
                    if current_run.status == "completed":
                        print("\n✓ Research complete! Rendering report...\n", flush=True)
                        return current_run.output_text
                    elif current_run.status in ["failed", "cancelled"]:
                        raise RuntimeError(f"Deep Research execution ended prematurely with status: {current_run.status}")
                    
                    elapsed = int(time.time() - start_time)
                    print(f"🔍 Agent has been searching and analyzing data for {elapsed} seconds...", end="\r", flush=True)
                    time.sleep(5)
            
            # Standard Responses API execution
            response_obj = client.responses.create(**kwargs)
            return response_obj.output_text
            
        except Exception:
            return _chat_completions_fallback(messages_context, model)
    else:
        try:
            return _chat_completions_fallback(messages_context, model)
        except Exception:
            kwargs = {
                "model": model,
                "input": messages_context,
                "max_output_tokens": 4096
            }
            if "deep-research" in model or "search" in model:
                kwargs["tools"] = [{"type": "web_search_preview"}]
            response_obj = client.responses.create(**kwargs)
            return response_obj.output_text


def _chat_completions_fallback(messages, model):
    """Internal helper targeting Chat Completions with safe token and temperature parameters."""
    kwargs = {
        "model": model,
        "messages": messages
    }
    
    is_reasoning = model.startswith("o") or "latest" in model or "mini" in model or "nano" in model or "4.1" in model
    is_search_preview = "search" in model

    if is_reasoning:
        kwargs["max_completion_tokens"] = 4096
    else:
        kwargs["max_tokens"] = 2048
        # Search preview models explicitly reject a rigid 0.0 temperature constraint
        if not is_search_preview:
            kwargs["temperature"] = 0.0

    completion = client.chat.completions.create(**kwargs)
    return completion.choices[0].message.content


def print_llm_response(prompt, model="gpt-5.5"):
    try:
        response = execute_llm_call(prompt, model=model)
        print(response)
    except Exception as e:
        print("Error:", str(e))


def get_llm_response(prompt, model="gpt-5.5"):
    try:
        return execute_llm_call(prompt, model=model)
    except Exception as e:
        return f"Error: {str(e)}"
    
    
def beautiful_barh(labels, values):
    plt.figure(figsize=(9, 5))
    bars = plt.barh(labels, values, color=plt.cm.tab20.colors)

    for bar in bars:
        plt.text(bar.get_width()/2,   
             bar.get_y() + bar.get_height()/2,  
             f'${bar.get_width() / 1e9:.1f}B',  
             ha='center', va='center', color='w', fontsize=10, fontweight="bold")
             
    def billions(x, pos):
        return f'${x * 1e-9:.1f}B'

    formatter = FuncFormatter(billions)
    plt.gca().xaxis.set_major_formatter(formatter)
    plt.gca().invert_yaxis()
